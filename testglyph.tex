\newcommand{\showglyph}[1]{%
  \directlua{\unexpanded{
    index = string.byte(tostring("#1"))-32
    tex.print("\\IfFileExists{glyph-" .. index .. ".mps}{\\includegraphics[width=\\digitwidth]{glyph-" .. index ..".mps} }{}")
  }}
}
\newcommand{\showtransition}[2]{%
  \subsection{Übergang #1…#2}
  \directlua{\unexpanded{
    from = string.byte(tostring("#1"))-32
    to = string.byte(tostring("#2"))-32
    res = from * 10000 + to * 100
    % print ("Hallo" , from, to )
    for index = res,res+99 do
         tex.print("\\IfFileExists{glyph-" .. index .. ".mps}{\\includegraphics[width=\\digitwidth]{glyph-" .. index ..".mps} }{}")
     end
  }}
  \hspace*{-1ex}
  \newpage
}
\newcommand{\showclock}[1]{%
  \IfFileExists{clock-#1.mps}{\includegraphics[width=\digitwidth]{clock-#1.mps} }{}
}
