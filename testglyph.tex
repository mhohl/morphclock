\newcommand{\showtransition}[2]{%
  \subsection{Übergang #1…#2}
  \directlua{\unexpanded{
    for index = 0,99 do
         if math.fmod(index+1,10) == 0 then
            maybenewline="\\newline"
         else
            maybenewline="\\hfill"
         end
         num = index
         if num < 10 then num = "0" .. num end
         tex.print("\\IfFileExists{mps/#1#2-" .. index .. ".mps}
                    {\\includegraphics[width=\\digitwidth]{mps/#1#2-" .. index ..".mps}←"
                    .. num .. maybenewline .. "}{}")
     end
  }}
  \hspace*{-1ex}

  \bigskip
  \hfill \hyperref[inhalt]{← zurück zum Inhaltsverzeichnis}
  \newpage
}

\newcommand{\showclocks}{%
  \directlua{\unexpanded{
     for index = 0,719 do
         if math.fmod(index+1,10) == 0 then
            maybenewline="\\newline"
         else
            maybenewline="\\hfill"
         end
         num = index
         if num < 10 then
            num ="00" .. num
         elseif num < 100 then
            num = "0" .. num
         end
         tex.print("\\IfFileExists{mps/clock-" .. index .. ".mps}
                    {\\includegraphics[width=\\digitwidth]{mps/clock-" .. index ..".mps}←"
                    .. num .. maybenewline .. "}{}")
     end
  }}
}
