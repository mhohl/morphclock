\newcommand{\showglyph}[1]{%
  \directlua{\unexpanded{
    index = string.byte(tostring("#1"))-32
    tex.print("\\IfFileExists{glyph-" .. index .. ".mps}{\\includegraphics[width=\\digitwidth]{glyph-" .. index ..".mps} }{}")
  }}
}
\newcommand{\showtransition}[2]{%
  \subsection{Übergang #1…#2}
  \directlua{\unexpanded{
    from = string.byte(tostring("#1"))-32
    to = string.byte(tostring("#2"))-32
    res = from * 10000 + to * 100
    % print ("Hallo" , from, to )
    for index = res,res+99 do
         if math.fmod(index+1,10) == 0 then
            maybenewline="\\newline"
         else
            maybenewline="\\hfill"
         end
         num = index-res
         if num < 10 then num = "0" .. num end
         tex.print("\\IfFileExists{glyph-" .. index .. ".mps}
                    {\\includegraphics[width=\\digitwidth]{glyph-" .. index ..".mps}←"
                    .. num .. maybenewline .. "}{}")
     end
  }}
  \hspace*{-1ex}

  \bigskip
  \hfill \hyperref[inhalt]{← zurück zum Inhaltsverzeichnis}
  \newpage
}
\newcommand{\showclock}[1]{%
  \IfFileExists{glyph-clock-#1.mps}{\includegraphics[width=\digitwidth]{glyph-clock-#1.mps}}{}
}
\newcommand{\showclocks}{%
  \directlua{\unexpanded{
     for index = 0,719 do
         if math.fmod(index+1,10) == 0 then
            maybenewline="\\newline"
         else
            maybenewline="\\hfill"
         end
         num = index
         if num < 10 then
            num ="00" .. num
         elseif num < 100 then
            num = "0" .. num
         end
         tex.print("\\IfFileExists{glyph-clock-" .. index .. ".mps}
                    {\\includegraphics[width=\\digitwidth]{glyph-clock-" .. index ..".mps}←"
                    .. num .. maybenewline .. "}{}")
     end
  }}
}
